## zh-CN

展示字数提示。

## en-US

Show character counting.

```jsx
import { Input } from 'antd';

const { TextArea } = Input;

const onChange = e => {
  console.log('Change:', e.target.value);
};

ReactDOM.render(
  <TextArea showCount maxLength={100} style={{ height: 120 }} onChange={onChange} />,
  mountNode,
);
```
